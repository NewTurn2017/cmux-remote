# M6 — APNs End-to-End

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make `cmux notification.created` events show up on the iPhone lock screen via Apple Push Notification service: relay-side JWT signer + APNs HTTP/2 client + EventForwarder fanout, iOS-side device-token registration + NotificationServiceExtension enrichment + deep-link routing.

**Architecture:** Relay holds one persistent `events.stream` subscription on cmux. `EventForwarder` filters notification events and dispatches a JSON payload to every device that has registered an `apns_token`. `APNsClient` signs an ES256 JWT (P8 key from disk), opens HTTP/2 to `api.push.apple.com:443` (or `api.sandbox.push.apple.com` in sandbox env), retries on transient failures, and clears `apns_token` from `DeviceStore` on `BadDeviceToken`. iOS registers for remote notifications, sends the device token to the relay via `POST /v1/devices/me/apns`, the NSE enriches incoming alerts with the workspace name, and tapping an alert opens `cmux://surface/<id>` which `DeepLinkRouter` resolves into the right view.

**Tech Stack:** swift-crypto (ES256 signing), async-http-client (HTTP/2), Foundation+UserNotifications on iOS, `NSEcontent` extension target via xcodegen.

**Branch:** `m6-apns` from `main`, after M5 has merged.

```swift
import Foundation
import Crypto

public final class APNsJWTSigner: @unchecked Sendable {
    public let keyId: String
    public let teamId: String
    private let privateKey: P256.Signing.PrivateKey
    private var cached: (token: String, expiresAt: Date)?
    private let lock = NSLock()

    public init(keyPEM: String, keyId: String, teamId: String) {
        self.privateKey = (try? P256.Signing.PrivateKey(pemRepresentation: keyPEM)) ?? P256.Signing.PrivateKey()
        self.keyId = keyId; self.teamId = teamId
    }

    public func token(at now: Date = Date()) throws -> String {
        lock.lock(); defer { lock.unlock() }
        if let c = cached, c.expiresAt > now.addingTimeInterval(5 * 60) { return c.token }
        let token = try signed(at: now)
        cached = (token, now.addingTimeInterval(60 * 60))
        return token
    }

    public func signed(at now: Date) throws -> String {
        let header: [String: String] = ["alg": "ES256", "kid": keyId, "typ": "JWT"]
        let payload: [String: Any] = ["iss": teamId, "iat": Int(now.timeIntervalSince1970)]
        let h = try Self.b64url(JSONSerialization.data(withJSONObject: header))
        let p = try Self.b64url(JSONSerialization.data(withJSONObject: payload))
        let signing = "\(h).\(p)"
        let sig = try privateKey.signature(for: Data(signing.utf8))
        let s = Self.b64url(sig.rawRepresentation)
        return "\(signing).\(s)"
    }

    private static func b64url(_ d: Data) -> String {
        d.base64EncodedString()
            .replacingOccurrences(of: "+", with: "-")
            .replacingOccurrences(of: "/", with: "_")
            .replacingOccurrences(of: "=", with: "")
    }
}

extension Data {
    public init?(base64URLEncoded s: String) {
        var v = s.replacingOccurrences(of: "-", with: "+")
                 .replacingOccurrences(of: "_", with: "/")
        while v.count % 4 != 0 { v.append("=") }
        guard let d = Data(base64Encoded: v) else { return nil }
        self = d
    }
}
```

```swift
import Foundation
import AsyncHTTPClient
import NIOCore

public protocol APNsTransport: Sendable {
    func post(url: String, headers: [String: String], body: Data) async throws -> (Int, Data)
}

public final class AsyncHTTPClientAPNsTransport: APNsTransport {
    public let httpClient: HTTPClient
    public init(httpClient: HTTPClient = HTTPClient(eventLoopGroupProvider: .singleton)) {
        self.httpClient = httpClient
    }
    public func post(url: String, headers: [String: String], body: Data) async throws -> (Int, Data) {
        var req = HTTPClientRequest(url: url)
        req.method = .POST
        for (k, v) in headers { req.headers.add(name: k, value: v) }
        req.body = .bytes(body)
        let resp = try await httpClient.execute(req, timeout: .seconds(5))
        let buf = try await resp.body.collect(upTo: 1 << 20)
        return (Int(resp.status.code), Data(buffer: buf))
    }
}

public enum APNsEnv: String { case prod, sandbox }

public enum APNsResult: Equatable {
    case delivered
    case badDeviceToken
    case retryable(Int)
    case fatal(String)
}

public final class APNsClient {
    public let transport: APNsTransport
    public let signerProvider: () throws -> String   // returns current JWT
    public let topic: String
    public let env: APNsEnv

    public init(transport: APNsTransport,
                signerProvider: @escaping () throws -> String,
                topic: String,
                env: APNsEnv)
    {
        self.transport = transport; self.signerProvider = signerProvider
        self.topic = topic; self.env = env
    }

    public func send(deviceToken: String, payload: Data,
                     collapseId: String?, threadId: String) async throws -> APNsResult
    {
        let host = env == .prod ? "api.push.apple.com" : "api.sandbox.push.apple.com"
        let url = "https://\(host)/3/device/\(deviceToken)"
        var headers: [String: String] = [
            "authorization": "bearer \(try signerProvider())",
            "apns-topic": topic,
            "apns-push-type": "alert",
            "apns-priority": "10",
        ]
        if let c = collapseId { headers["apns-collapse-id"] = c }
        let (status, body) = try await transport.post(url: url, headers: headers, body: payload)
        switch status {
        case 200: return .delivered
        case 410: return .badDeviceToken
        case 429, 500...599: return .retryable(status)
        default:
            let s = String(data: body, encoding: .utf8) ?? ""
            if s.contains("BadDeviceToken") { return .badDeviceToken }
            return .fatal(s)
        }
    }
}
```

```swift
import Foundation
import SharedKit

public actor APNsSender {
    public let client: APNsClient
    public let deviceStore: DeviceStore
    public init(client: APNsClient, deviceStore: DeviceStore) {
        self.client = client; self.deviceStore = deviceStore
    }

    public func send(notification n: NotificationRecord, to devices: [Device]) async {
        let payload = Self.makePayload(for: n)
        for dev in devices where dev.apnsToken != nil {
            await sendOne(deviceId: dev.deviceId, token: dev.apnsToken!, payload: payload, threadId: n.threadId)
        }
    }

    private func sendOne(deviceId: String, token: String, payload: Data, threadId: String) async {
        var attempts = 0
        while attempts < 3 {
            attempts += 1
            do {
                switch try await client.send(deviceToken: token, payload: payload,
                                             collapseId: nil, threadId: threadId) {
                case .delivered: return
                case .badDeviceToken:
                    try? deviceStore.clearAPNsToken(deviceId: deviceId); return
                case .retryable: try? await Task.sleep(nanoseconds: UInt64(0.5 * Double(attempts) * 1e9))
                case .fatal: return
                }
            } catch {
                try? await Task.sleep(nanoseconds: UInt64(0.5 * Double(attempts) * 1e9))
            }
        }
    }

    public static func makePayload(for n: NotificationRecord) -> Data {
        let aps: [String: Any] = [
            "alert": [
                "title": n.title,
                "subtitle": n.subtitle ?? "",
                "body": n.body,
            ],
            "thread-id": n.threadId,
            "sound": "default",
            "mutable-content": 1,
            "interruption-level": "active",
        ]
        var cmux: [String: Any] = ["kind": "notification.created",
                                    "workspace_id": n.workspaceId,
                                    "notification_id": n.id]
        if let s = n.surfaceId { cmux["surface_id"] = s }
        let root: [String: Any] = ["aps": aps, "cmux": cmux]
        return (try? JSONSerialization.data(withJSONObject: root, options: [])) ?? Data()
    }
}
```

```swift
import Foundation
import SharedKit

public actor EventForwarder {
    public let sender: APNsSender
    public let sessionManager: SessionManager
    public let deviceStore: DeviceStore
    public let allowedCategories: Set<EventCategory>

    public init(sender: APNsSender, sessionManager: SessionManager,
                deviceStore: DeviceStore, allowedCategories: Set<EventCategory>)
    {
        self.sender = sender; self.sessionManager = sessionManager
        self.deviceStore = deviceStore; self.allowedCategories = allowedCategories
    }

    public func ingest(_ event: EventFrame) async {
        guard allowedCategories.contains(event.category) else { return }
        // 1. Always broadcast to all live WS sessions for in-app updates.
        await sessionManager.broadcastToAll(frame: .event(event))
        // 2. If it's a notification.created, send APNs.
        guard event.name == "notification.created" else { return }
        guard let data = try? JSONEncoder().encode(event.payload),
              let n = try? JSONDecoder().decode(NotificationRecord.self, from: data)
        else { return }
        await sender.send(notification: n, to: deviceStore.allDevices())
    }
}
```

```swift
import UIKit
import UserNotifications

public final class PushBootstrap {
    private let host: String
    private let port: Int
    private let keychain: Keychain
    private let http: HTTPClientFacade
    public init(host: String, port: Int, keychain: Keychain, http: HTTPClientFacade) {
        self.host = host; self.port = port; self.keychain = keychain; self.http = http
    }

    public func requestAuth() async {
        let center = UNUserNotificationCenter.current()
        _ = try? await center.requestAuthorization(options: [.alert, .badge, .sound])
        await MainActor.run { UIApplication.shared.registerForRemoteNotifications() }
    }

    public func register(token: Data) async {
        guard let bearer = try? keychain.get("bearer") else { return }
        let hex = token.map { String(format: "%02x", $0) }.joined()
        var req = URLRequest(url: URL(string: "https://\(host):\(port)/v1/devices/me/apns")!)
        req.httpMethod = "POST"
        req.setValue("Bearer \(bearer)", forHTTPHeaderField: "Authorization")
        req.setValue("application/json", forHTTPHeaderField: "Content-Type")
        let env = (Bundle.main.object(forInfoDictionaryKey: "APNS_ENV") as? String) ?? "prod"
        req.httpBody = Data(#"{"apns_token":"\#(hex)","env":"\#(env)"}"#.utf8)
        _ = try? await http.request(req)
    }
}

@MainActor
public final class AppDelegate: NSObject, UIApplicationDelegate {
    public var pushBootstrap: PushBootstrap?

    public func application(_ application: UIApplication,
        didRegisterForRemoteNotificationsWithDeviceToken deviceToken: Data)
    {
        Task { await pushBootstrap?.register(token: deviceToken) }
    }
    public func application(_ application: UIApplication,
        didFailToRegisterForRemoteNotificationsWithError error: Error)
    {
        // best-effort; user can retry from Settings
    }
}
```

```swift
import UserNotifications

final class NotificationService: UNNotificationServiceExtension {
    var contentHandler: ((UNNotificationContent) -> Void)?
    var working: UNMutableNotificationContent?

    override func didReceive(_ request: UNNotificationRequest,
                             withContentHandler handler: @escaping (UNNotificationContent) -> Void)
    {
        self.contentHandler = handler
        guard let mutable = request.content.mutableCopy() as? UNMutableNotificationContent else {
            handler(request.content); return
        }
        self.working = mutable
        if let cmux = request.content.userInfo["cmux"] as? [String: Any],
           let wsId = cmux["workspace_id"] as? String,
           let wsName = wsName(forId: wsId)
        {
            mutable.subtitle = "ws/\(wsName)"
        }
        handler(mutable)
    }

    override func serviceExtensionTimeWillExpire() {
        if let h = contentHandler, let w = working { h(w) }
    }

    private func wsName(forId id: String) -> String? {
        // Look up the cached workspaces written by the app via App Group container.
        let group = "group.com.genie.cmuxremote"
        let url = FileManager.default.containerURL(forSecurityApplicationGroupIdentifier: group)?
            .appendingPathComponent("workspaces.json")
        guard let url, let data = try? Data(contentsOf: url),
              let arr = try? JSONDecoder().decode([WS].self, from: data),
              let row = arr.first(where: { $0.id == id })
        else { return nil }
        return row.name
    }
    struct WS: Decodable { let id: String; let name: String }
}
```

```swift
import Foundation

public enum DeepLinkRoute: Equatable {
    case surface(id: String)
    case workspace(id: String)
    case unknown
}

public enum DeepLinkRouter {
    public static func parse(_ url: URL) -> DeepLinkRoute {
        guard url.scheme == "cmux" else { return .unknown }
        let host = url.host ?? ""
        let id = url.pathComponents.dropFirst().first ?? ""
        switch host {
        case "surface":   return id.isEmpty ? .unknown : .surface(id: id)
        case "workspace": return id.isEmpty ? .unknown : .workspace(id: id)
        default:          return .unknown
        }
    }
}
```

- [ ] **Step 3: Wire into `CmuxRemoteApp.handleDeepLink`**

